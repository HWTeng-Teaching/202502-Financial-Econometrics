
## a
```
# 線性模型
EXPER <- 0:30
RATING_hat <- 64.289 + 0.990 * EXPER

plot(EXPER, RATING_hat, type = "l", col = "blue", lwd = 2,
     xlab = "EXPER", ylab = "RATING",
     main = "Model 1")
points(EXPER, RATING_hat, col = "blue", pch = 16) # 標記數據點
grid()
```

# 對數模型
EXPER <- 1:30
RATING_hat <- 39.464 + 15.312 * log(EXPER)
