cps5_small$educ2 <- cps5_small$educ^2

quad_model <- lm(wage ~ educ2, data = cps5_small)

summary(quad_model)

alpha2 <- coef(quad_model)["educ2"]

# 計算 12 年教育的邊際影響
educ_12 <- 12
ME_12 <- 2 * alpha2 * educ_12

# 計算 16 年教育的邊際影響
educ_16 <- 16
ME_16 <- 2 * alpha2 * educ_16

cat("邊際影響 (12 年教育):", ME_12, "\n")
cat("邊際影響 (16 年教育):", ME_16, "\n")
