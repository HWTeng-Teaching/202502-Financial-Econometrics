    threshold_lev <- (2 * p) / n
    which(lev > threshold_lev)
    
    # (3) DFBETAS：判斷哪些觀測值對係數影響大
    dfb <- dfbetas(model3)

    which(abs(dfb[,2]) > 1)  # 看 TIME² 的影響
    
    # (4) DFFITS：判斷對整體擬合影響大的觀測值
    dff <- dffits(model3)
    threshold_dffits <- 2 * sqrt(p / n)
    which(abs(dff) > threshold_dffits)
